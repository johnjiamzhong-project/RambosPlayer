🎵 音频混合架构 — RambosPlayer

工具 → 音频混合：将本地音频或实时录音按可调比例混入源视频音频，支持多段叠加、试播放和 Timeline 可视化。

3. avfilter 滤镜图详解

滤镜字符串模板（N 个区间）

// 源音频：各区间时间窗内降低音量
[in0]volume=srcVol1:enable='between(t,s1,e1)',
     volume=srcVol2:enable='between(t,s2,e2)',...[src_adj];

// 各区间附加音频：截取 → 重置PTS → 延迟 → 补静音 → 音量
[in1]atrim=start=off1:end=off1+dur1,asetpts=PTS-STARTPTS,
     adelay=videoStartMs1|videoStartMs1,apad,volume=mixVol1[add1];

[in2]atrim=...,adelay=...,apad,volume=...[add2];

// amix：直接叠加，输出时长 = 源音频
[src_adj][add1][add2]...amix=inputs=N+1:normalize=0:duration=first[out]

4. 核心数据结构

struct AudioMixRegion {
    QString  sourcePath;       // 音频文件路径（录音为临时 WAV）
    QString  displayName;      // 列表显示名
    int64_t  videoStartUs;     // 贴入视频的时间（微秒）
    int64_t  audioOffsetUs;    // 从音频文件哪里开始（微秒）
    int64_t  audioDurationUs;  // 使用多少时长（微秒）
    float    srcVol;           // 该区间源视频音量（0.0–1.0）
    float    mixVol;           // 新增音频音量，srcVol + mixVol = 1.0
    bool     isRecorded;       // true = 录音临时文件，析构时删除
};

struct AudioMixTask {
    QString               originalVideoPath;  // 始终为原始源视频
    QList<AudioMixRegion> regions;            // 音频区间列表
    QString               outputPath;         // 输出 MP4 路径
};
持续混合策略：sourceFile_ 始终存储原始源视频路径；每次导出都从原始源重新混合所有累积区间，避免多代 AAC 编解码质量损失。

7. Slider 联动约束

srcVol + mixVol = 100%，两个 Slider 互补联动：

void AudioMixPanel::onSrcVolChanged(int value) {
    int mixVal = 100 - value;
    ui->mixVolSlider->blockSignals(true);
    ui->mixVolSlider->setValue(mixVal);
    ui->mixVolSlider->blockSignals(false);
}

void AudioMixPanel::onMixVolChanged(int value) {
    int srcVal = 100 - value;
    ui->srcVolSlider->blockSignals(true);
    ui->srcVolSlider->setValue(srcVal);
    ui->srcVolSlider->blockSignals(false);
}
